## Usage

```tsx
import { Resizable, ResizablePanel, ResizableHandle } from "@/components/ui/resizable";
```

```tsx
<Resizable>
  <ResizablePanel>A</ResizablePanel>
  <ResizableHandle />
  <ResizablePanel>B</ResizablePanel>
</Resizable>
```
