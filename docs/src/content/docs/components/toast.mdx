## Usage

```tsx
import { toaster } from "@kobalte/core";
import {
  Toast,
  ToastContent,
  ToastProgress,
  ToastTitle
} from "@/components/ui/toast";
```

```tsx
toaster.show(props => (
  <Toast toastId={props.toastId}>
    <ToastContent>
      <ToastTitle>Toast</ToastTitle>
    </ToastContent>
    <ToastProgress />
  </Toast>
))
```

```tsx title="src/main.tsx" {1,11-13}
import { ToastRegion, ToastList } from "@/components/ui/toast"
 
const App = () => {
  return (
    <Router
      root={props => (
          <Suspense>
            <ColorModeScript />
            <ColorModeProvider>
              {props.children}
              <ToastRegion>
                <ToastList />
              </ToastRegion>
            </ColorModeProvider>
          </Suspense>
      )}
    >
      <FileRoutes />
    </Router>
  );
};
 
export default App;
```
