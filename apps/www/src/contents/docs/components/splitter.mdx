## Usage

```tsx
import { Splitter, SplitterPanel, SplitterResizeTrigger } from "@/components/ui/splitter";
```

```tsx
<Splitter
  size={[
    { id: "1", size: 50 },
    { id: "2", size: 50 }
  ]}
>
  <SplitterPanel id="1">A</SplitterPanel>
  <SplitterResizeTrigger id="1:2" />
  <SplitterPanel id="2">B</SplitterPanel>
</Splitter>
```
