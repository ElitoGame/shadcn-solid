<ComponentSource name="sonner" />

<Step>Add the Toaster component</Step>

```tsx title="src/app.tsx" {1,11}
import { Toaster } from "@/components/ui/sonner"

const App = () => {
  return (
    <Router
      root={props => (
          <Suspense>
            <ColorModeScript />
            <ColorModeProvider>
              {props.children}
              <Toaster />
            </ColorModeProvider>
          </Suspense>
      )}
    >
      <FileRoutes />
    </Router>
  );
};

export default App;
```
