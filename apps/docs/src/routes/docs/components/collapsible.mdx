## Installation

1. Install the `@kobalte/core` component from kobalte:

```bash
pnpm add @kobalte/core
```

2. Copy and paste the following code into your project.

<ComponentSource src="/src/components/ui/collapsible.tsx" />

## uno.config.ts

Add the following animations to your `uno.config.ts` file:

```tsx title="uno.config.ts" {3-18}
export default defineConfig({
	theme: {
		animation: {
			keyframes: {
				"collapsible-down":
					"{ from { height: 0 } to { height: var(--kb-collapsible-content-height) }}",
				"collapsible-up":
					"{ from { height: var(--kb-collapsible-content-height) } to { height: 0 }}",
			},
			durations: {
				"collapsible-down": "200ms",
				"collapsible-up": "200ms",
			},
			timingFns: {
				"collapsible-down": "ease-out",
				"collapsible-up": "ease-out",
			},
		},
	},
})
```

## Usage

```tsx
import {
	Button,
	Collapsible,
	CollapsibleContent,
	CollapsibleTrigger,
} from "~/components"
```

```tsx
<Collapsible>
	<CollapsibleTrigger>Can I use this in my project?</CollapsibleTrigger>
	<CollapsibleContent>
		Yes. Free to use for personal and commercial projects. No attribution
		required.
	</CollapsibleContent>
</Collapsible>
```
