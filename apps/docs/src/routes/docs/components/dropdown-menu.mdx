## Installation

1. Install the `@kobalte/core` component from kobalte:

```bash
pnpm add @kobalte/core
```

2. Copy and paste the following code into your project.

<ComponentSource src="/src/components/ui/dropdown-menu.tsx" />

## uno.config.ts

Add the following animations to your `uno.config.ts` file:

```tsx title="uno.config.ts" {3-21}
export default defineConfig({
	theme: {
		animation: {
			keyframes: {
				"dropdown-show":
					"{ from { opacity: 0; transform: scale(0.96) } to { opacity: 1; transform: scale(1) }}",
				"dropdown-hide":
					"{ from { opacity: 1; transform: scale(1) } to { opacity: 0; transform: scale(0.96) }}",
			},
			durations: {
				"dropdown-show": "200ms",
				"dropdown-hide": "200ms",
			},
			timingFns: {
				"dropdown-show": "ease-out",
				"dropdown-hide": "ease-in",
			},
			counts: {
				"dropdown-hide": "forwards",
			},
		},
	},
})
```

## Usage

```tsx
import {
	DropdownMenu,
	DropdownMenuContent,
	DropdownMenuItem,
	DropdownMenuLabel,
	DropdownMenuSeparator,
	DropdownMenuTrigger,
} from "@/components/ui/dropdown-menu"
```

```tsx
<DropdownMenu>
	<DropdownMenuTrigger>Open</DropdownMenuTrigger>
	<DropdownMenuContent>
		<DropdownMenuLabel>My Account</DropdownMenuLabel>
		<DropdownMenuSeparator />
		<DropdownMenuItem>Profile</DropdownMenuItem>
		<DropdownMenuItem>Billing</DropdownMenuItem>
		<DropdownMenuItem>Team</DropdownMenuItem>
		<DropdownMenuItem>Subscription</DropdownMenuItem>
	</DropdownMenuContent>
</DropdownMenu>
```
